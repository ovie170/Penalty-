  <div class="scoreboard">Score: <span id="score">0</span></div>
  <div id="game" onclick="kick(event)">
    <div class="goal"></div>
    <div class="goalie" id="goalie"></div>
    <div class="ball" id="ball"></div>
  </div>

  <script>
    let score = 0;
    const goalie = document.getElementById('goalie');
    const ball = document.getElementById('ball');
    const scoreDisplay = document.getElementById('score');

    function moveGoalie() {
      const left = Math.random() * 320;
      goalie.style.left = `${left}px`;
    }

    function kick(event) {
      const game = document.getElementById('game');
      const x = event.clientX - game.getBoundingClientRect().left;
      const y = event.clientY - game.getBoundingClientRect().top;

      // Animate ball to where user tapped
      ball.style.left = `${x - 15}px`;
      ball.style.bottom = `${600 - y}px`;

      setTimeout(() => {
        // Collision detection
        const goalieLeft = goalie.offsetLeft;
        const ballLeft = ball.offsetLeft;

        if (ballLeft > goalieLeft - 30 && ballLeft < goalieLeft + 80) {
          alert("😢 Missed! Try again.");
          score = 0;
        } else {
          alert("🎉 Goal!");
          score++;
        }

        scoreDisplay.textContent = score;
        resetBall();
        moveGoalie();
      }, 500);
    }

    function resetBall() {
      ball.style.bottom = "40px";
      ball.style.left = "50%";
    }

    moveGoalie();
    setInterval(moveGoalie, 2000);
  </script>
